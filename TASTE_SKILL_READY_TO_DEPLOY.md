```jsx
import { 
  PremiumCard, 
  PremiumButton, 
  PremiumMetric,
  PremiumChart,
  PremiumAlert 
} from '../components/premium';

import { 
  tastePatterns, 
  motionPresets 
} from '../plugins/taste-plugin';
```

### Use PremiumCard
```jsx
<PremiumCard variant="dark" interactive>
  <h3>Premium Card</h3>
  <p>Content here</p>
</PremiumCard>
```

### Use PremiumButton
```jsx
<PremiumButton variant="primary" onClick={handleClick}>
  Click Me
</PremiumButton>
```

### Use PremiumMetric
```jsx
<PremiumMetric
  label="Temperature"
  value="28°C"
  trend={5}
  icon={Cloud}
/>
```

```jsx
<div className={tastePatterns.premiumCard.base}>
  Custom card using patterns
</div>
```

### Use Animations
```jsx
<motion.div {...motionPresets.fadeInUp}>
  Animated content
</motion.div>
```
